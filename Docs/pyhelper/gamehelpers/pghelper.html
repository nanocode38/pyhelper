 46def disassemble_sprite_sheet(image_path: str, width: int, height: int, num_of_image) -> list:
 47    """
 48    This function processes a Sprite image into a set of loaded Surface objects
 49
 50    Args:
 51        image_path: the path to the total Sprite graph
 52        width: The length of an image
 53        height: The height of an image
 54        num_of_image: The total number of photos
 55
 56    Returns:
 57        list: Surface objects for all images
 58    """
 59    images = []
 60    sprite_sheet_image = pygame.image.load(image_path)
 61    n_cols = sprite_sheet_image.get_width() // width
 62
 63    row, col = 0, 0
 64    for image_num in range(num_of_image):
 65        x = col * height
 66        y = row * width
 67
 68        # Create a subsurface from the bigger spriteSheet
 69        subsurface_rect = pygame.Rect(x, y, width, height)
 70        image = sprite_sheet_image.subsurface(subsurface_rect)
 71        images.append(image)
 72
 73        col += 1
 74        if col == n_cols:
 75            col = 0
 76            row += 1
 77
 78    return images

 97class Background:
 98    """
 99    Draw the game background
100
101    Args:
102        screen: the Surface of the scene to paint
103        image_path: image path for background image
104        width: The background width, default: 0, the original width, can be pygame.FULLSCREEN is the screen width
105        height: The background height, default: 0, the original height, can be pygame.FULLSCREEN is the screen height
106    """
107
108    def __init__(self):
109        self.image = None
110
111    def __call__(
112        self,
113        screen: pygame.SurfaceType,
114        image_path: str,
115        width: int = 0,
116        height: int = 0,
117    ) -> None:
118        if self.image is None:
119            self.image = pygame.image.load(image_path)
120            if width != 0:
121                if width == pygame.FULLSCREEN:
122                    width = screen.get_width()
123                self.image = pygame.transform.scale(self.image, (width, self.image.get_height()))
124            if height != 0:
125                if height == pygame.FULLSCREEN:
126                    height = screen.get_height()
127                self.image = pygame.transform.scale(self.image, (self.image.get_width(), height))
128        screen.blit(self.image, (0, 0))

135class Scene(ABC):
136    """
137    The Scene class is an abstract class to be used as a base class for any scenes that you want to create.
138
139    In the startup code for your  application, you create an instance of each of the
140    scenes that you want to be available.  Then you build a dictionary like this
141
142        |    {<sceneKey1>:<sceneObject1>, <sceneKey2>:<sceneObject2>, ...}
143
144    Then you pass this dictionary when you instantiate the Scene Manager.
145    See the SceneMgr documentation for details.
146
147    To create a scene, you must write a subclass that inherits from this class.
148    Your class must implement these methods
149
150        |   __init__(), update(), draw(), get_scene_key()
151
152    Other methods can be overridden if you wish:
153
154        | enter(), leave()
155
156    You can add any additional methods that your class requires.
157    Additionally, there are other methods to allow you to get or set info, or navigate to other scenes:
158
159        | go_to_scene(), quit(), request(), respond() and more.
160
161    In the __init__() method of your scene subclass, you will receive a window reference.
162    You should copy this into an instance variable like this, and use it when drawing:
163
164        |    def __init__(self, screen):
165        |        self.screen = screen
166        |        self.screen_rect = screen.get_rect()
167        |        # Add any initialization you want to do here.
168
169
170        You also need to write a get_scene_key() method that returns a string
171        or constant that uniquely identifies the scene.  It is recommended that you
172        build and import a Constants.py file that contains constants for each scene,
173        and use the key associated with the current scene here.
174        |    def get_scene_key(self):
175        |        return <string or CONSTANT that identifies this scene>
176
177    When your scene is active, the SceneManager calls a standard set of methods in the current scene.
178    Therefore, all scenes must implement these methods (polymorphism):
179
180        |    update()        # called in every frame
181        |    draw()          # called in every frame
182
183    The following methods can optionally be implemented in a scene.  If they are not
184    implemented, then the null version in the Scene subclass will be used.
185    (The Scene class' default versions only execute a pass statement):
186
187        |    enter()          # called once whenever the scene is entered
188        |    update()         # called in every frame
189        |    leave()          # called once whenever the scene is left
190
191    When you want to go to a new scene, call:
192
193        |    self.go_to_scene() and pass in the sceneKey of the scene you want to go to,
194        |    and optionally, pass any data you want the next scene to receive in its enter() method.
195
196    If you want to quit the program from your scene, call:
197        |    self.quit()
198    """
199

201    def __init__(self, screen: pygame.Surface):
202        self.screen = screen
203        self.screen_rect = screen.get_rect()

207        """This returns a unique identity string that represents the internal representation of the scene"""
208        pass

214
215        Args:
216            events: a list of events your method should handle.
217            key_pressed_list: a list of keys that are pressed (a Boolean for each key).
218        """
219        pass

222        """
223        This method is called whenever the user enters a scene.
224        Should be overridden if you expect data when your scene is entered.
225        Add any code you need to start or re-start the scene
226
227        Args:
228            data: can be of any type agreed to by the old and new scenes
229        """
230        pass

259            next_scene_key: the scene key (string) of the scene to go to
260            data: any data you want sent to the next scene (defaults: None)
261        """

291    def send_all(self, send_id, info) -> None:
292        """
293        Call this method to send information to all other scenes
294
295        The other scenes must implement a method named:  receive().
296        You can pass any info that the sender and all other scenes agree upon
297
298        Args:
299            send_id: the type of data you are sending the target scene
300            info: the actual data to send
301        """
302        self.scene_mgr._send_all_receive(self, send_id, info)  # pass in self to identify sender
303
304    def respond(self, request_id: Any) -> None:
305        """
306        Respond to a request for information from some other scene
307
308        You must override this method if your scene expects to handle
309        requests for information from other scenes via calls to:  request()
310        Args:
311            request_id: identifier of what data to be sent back to the caller
312        """
313        raise NotImplementedError
314
315    def receive(self, receive_id: Any, info: Any) -> None:
316        """
317        Receives information from another scene.
318
319        You must override this method if your scene expects to receive information from
320        other scenes sending information via calls to:  send()
321
322        Args:
323            receive_id - an identifier for what type of information is being received
324            info - the information sent from another scene
325        """
326        raise NotImplementedError
327
328    def add_scene(self, scene_key: str, scene) -> None:
329        """
330        Call this method whenever you want to add a new scene dynamically.
331        For example, you could have a game with many levels (each implemented as a scene),
332        but only have the current level loaded.  As the player completes a level, you
333        could do a remove_scene on the current level and do an add_scene on the
334        next level, then do a goToScene on the new level.
335
336        Args:
337            scene_key: a key to uniquely identify this scene
338            scene: an instance of the new scene to be added
339            (typically, you would instantiate the new scene, and pass in that reference to this call)
340        """
341        self.scene_mgr._add_scene(scene_key, scene)
342
343    def remove_scene(self, scene_key: str) -> None:
344        """
345        Call this method whenever you want to remove an existing scene
346        You can remove a scene to save memory - the scene object will be deleted.
347        The SceneMgr delays removing the scene until the next time through the main loop.
348        Therefore, it is safe to call to remove_scene from its own scene, if you immediately
349        go to another scene.
350
351        Args:
352            scene_key: the scene key (string) of the scene to remove"""
353        self.scene_mgr._remove_scene(scene_key)

356class SceneMgr:
357    """
358    SceneMgr (Scene Manager)  allows you to build a program with multiple scenes.
359
360    The SceneMgr manages any number of scenes built as subclasses of the "Scene" class.
361    For more details, see the "Scene" class.
362
363    Args:
364        scenes: is a dictionary or list that consists of:
365            {<sceneKey>: <sceneObject>, <sceneKey>: <sceneObject>, ...} or [<sceneObject>, <sceneObject>, ...]
366              where each sceneObject is an object instantiated from a scene class
367              (For details on Scenes, see the Scene class)
368        fps - is the frames per second at which the program should run
369
370    Based on the concept of a "Scene Manager" by Blake O'Hare of Nerd Paradise
371    """
372
373    def __init__(
374        self,
375        scenes: Union[list, dict],
376        fps: int,
377        frame_rate_display: "widgets.DisplayText | None" = None,
378    ):
379        if isinstance(scenes, dict):
380            self.scenes_dict = scenes
381            keys_list = list(self.scenes_dict)  # get all the keys
382            starting_key = keys_list[0]  # first key is the starting scene ley
383            self.current_scene = self.scenes_dict[starting_key]
384
385        else:  # Older style, we start with a list of scenes
386            # Build a dictionary, each _entry of which is a scene key : scene object
387            # Then we call getSceneKey so each scene can identify itself
388            self.scenes_dict = {}
389            for scene in scenes:
390                key = scene.get_scene_key()  # Each scene must return a unique key to identify itself
391                self.scenes_dict[key] = scene
392            # The first element in the list is the used as the starting scene
393            self.current_scene = scenes[0]
394
395        self.fps = fps
396        self.frame_rate_display = frame_rate_display
397        self.show_frame_rate = frame_rate_display is not None  # for fast checking in main loop
398        self.scenes_to_remove_list = []
399
400        # Give each scene a reference back to the SceneMgr.
401        # This allows any scene to do a goToScene, request, send,
402        # or send_all, which gets forwarded to the scene manager.
403        for key, scene in self.scenes_dict.items():
404            scene._set_ref_to_scene_mgr(self)
405
406    def run(self) -> None:
407        """
408        This method implements the main pygame loop.
409        It should typically be called as the last line of your main program.
410        It is designed to call a standardized set of methods in the current scene.
411        All scenes must implement the following methods (polymorphism):
412        |   update()  # called in every frame
413        |   draw()          # called in every frame
414
415        The following methods can be implemented in a scene:
416            |   enter()  # called once whenever the scene is entered
417            |   leave()  # called once whenever the scene is left
418
419        If any is not implemented, then the version in the Scene base class,
420        which only performs a pass statement, will be used.
421        """
422
423        clock = pygame.time.Clock()
424
425        # 6 - Loop forever
426        while True:
427            # If there are any scenes to be removed (keys added by remove_scene method)
428            if not (self.scenes_to_remove_list is []):
429                for key in self.scenes_to_remove_list:
430                    del self.scenes_dict[key]
431                self.scenes_to_remove_list = []  # reset
432
433            keys_down_list = pygame.key.get_pressed()
434
435            # 7 - Check for and handle events
436            events_list = []
437            for event in pygame.event.get():
438                if (event.type == pygame.QUIT) or ((event.type == pygame.KEYDOWN) and (event.key == pygame.K_ESCAPE)):
439                    # Tell current scene we're leaving
440                    self.current_scene.leave()
441                    pygame.quit()
442                    sys.exit()
443
444                events_list.append(event)
445
446            # Here, we let the current scene process all events by calling its handleInputs() method
447            # do any "per frame" actions in its update() method,
448            # and call its draw() method so it can draw everything that needs to be drawn.
449            self.current_scene.update(events_list, keys_down_list)
450            self.current_scene.draw()
451            if self.show_frame_rate:
452                fps = str(clock.get_fps())
453                self.frame_rate_display.set_value("FPS: " + fps)
454                self.frame_rate_display.draw()
455
456            # 11 - Update the window
457            pygame.display.update()
458
459            # 12 - Slow things down a bit
460            clock.tick(self.fps)
461
462    def _go_to_scene(self, next_scene_key, data_for_next_scene):
463        if next_scene_key is None:  # meaning, exit
464            pygame.quit()
465            sys.exit()
466
467        # Call the leave method of the old scene to allow it to clean up.
468        # Set the new scene (based on the key) and
469        # call the enter method of the new scene.
470        self.current_scene.leave()
471        pygame.key.set_repeat(0)  # turn off repeating characters
472        try:
473            self.current_scene = self.scenes_dict[next_scene_key]
474        except KeyError:
475            raise KeyError(
476                "Trying to go to scene '" + next_scene_key + "' but that key is not in the dictionary of scenes."
477            )
478        self.current_scene.enter(data_for_next_scene)
479
480    def _request_respond(self, target_scene_key, request_id):
481        target_scene = self.scenes_dict[target_scene_key]
482        info = target_scene.respond(request_id)
483        return info
484
485    def _send_receive(self, target_scene_key, send_id, info):
486        target_scene = self.scenes_dict[target_scene_key]
487        target_scene.receive(send_id, info)
488
489    def _send_all_receive(self, sender_scene, send_id, info):
490        for scene_key in self.scenes_dict:
491            target_scene = self.scenes_dict[scene_key]
492            if target_scene != sender_scene:
493                target_scene.receive(send_id, info)
494
495    def _add_scene(self, new_scene):
496        # Ask the new scene for its scene key, and add it to the scenes dictionary
497        new_scene_key = new_scene.getSceneKey()
498        if new_scene_key in self.scenes_dict:
499            raise KeyError("Trying to add a scene with key" + new_scene_key + "but that scene key already exists")
500        self.scenes_dict[new_scene_key] = new_scene
501        # Send the new scene a reference to the SceneMgr
502        new_scene._set_ref_to_scene_mgr(self)
503
504    def _remove_scene(self, scene_key_to_remove):
505        if not (scene_key_to_remove in self.scenes_dict):
506            raise KeyError(
507                "Attempting to remove scene with key "
508                + scene_key_to_remove
509                + " but no scene with that key currently exists (either never defined or removed)."
510            )
511        # Add the key to a list of keys to be removed
512        # The scene(s) will be removed the next time around the main loop.from
513        # This allows a scene to make a call to remove_scene to remove itself right before doing a goToScene
514        #    (and the call will return successfully)
515        self.scenes_to_remove_list.append(scene_key_to_remove)

The Scene class is an abstract class to be used as a base class for any scenes that you want to create.

In the startup code for your application, you create an instance of each of the scenes that you want to be available. Then you build a dictionary like this

|    {<sceneKey1>:<sceneObject1>, <sceneKey2>:<sceneObject2>, ...}

Then you pass this dictionary when you instantiate the Scene Manager. See the SceneMgr documentation for details.

To create a scene, you must write a subclass that inherits from this class. Your class must implement these methods

|   __init__(), update(), draw(), get_scene_key()

Other methods can be overridden if you wish:

| enter(), leave()

You can add any additional methods that your class requires. Additionally, there are other methods to allow you to get or set info, or navigate to other scenes:

| go_to_scene(), quit(), request(), respond() and more.

In the __init__() method of your scene subclass, you will receive a window reference. You should copy this into an instance variable like this, and use it when drawing:

|    def __init__(self, screen):
|        self.screen = screen
|        self.screen_rect = screen.get_rect()
|        # Add any initialization you want to do here.


You also need to write a get_scene_key() method that returns a string
or constant that uniquely identifies the scene.  It is recommended that you
build and import a Constants.py file that contains constants for each scene,
and use the key associated with the current scene here.
|    def get_scene_key(self):
|        return <string or CONSTANT that identifies this scene>

When your scene is active, the SceneManager calls a standard set of methods in the current scene. Therefore, all scenes must implement these methods (polymorphism):

|    update()        # called in every frame
|    draw()          # called in every frame

The following methods can optionally be implemented in a scene. If they are not implemented, then the null version in the Scene subclass will be used. (The Scene class' default versions only execute a pass statement):

|    enter()          # called once whenever the scene is entered
|    update()         # called in every frame
|    leave()          # called once whenever the scene is left

When you want to go to a new scene, call:

|    self.go_to_scene() and pass in the sceneKey of the scene you want to go to,
|    and optionally, pass any data you want the next scene to receive in its enter() method.

If you want to quit the program from your scene, call: | self.quit()

412        All scenes must implement the following methods (polymorphism):
413        |   update()  # called in every frame
414        |   draw()          # called in every frame
415

422        """
423
424        clock = pygame.time.Clock()
425
426        # 6 - Loop forever
427        while True:
428            # If there are any scenes to be removed (keys added by remove_scene method)
429            if not (self.scenes_to_remove_list is []):
430                for key in self.scenes_to_remove_list:
431                    del self.scenes_dict[key]
432                self.scenes_to_remove_list = []  # reset
433
434            keys_down_list = pygame.key.get_pressed()
435
436            # 7 - Check for and handle events
437            events_list = []
438            for event in pygame.event.get():
439                if (event.type == pygame.QUIT) or ((event.type == pygame.KEYDOWN) and (event.key == pygame.K_ESCAPE)):
440                    # Tell current scene we're leaving
441                    self.current_scene.leave()
442                    pygame.quit()
443                    sys.exit()
444
445                events_list.append(event)
446
447            # Here, we let the current scene process all events by calling its handleInputs() method
448            # do any "per frame" actions in its update() method,
449            # and call its draw() method so it can draw everything that needs to be drawn.
450            self.current_scene.update(events_list, keys_down_list)
451            self.current_scene.draw()
452            if self.show_frame_rate:
453                fps = str(clock.get_fps())
454                self.frame_rate_display.set_value("FPS: " + fps)
455                self.frame_rate_display.draw()
456
457            # 11 - Update the window
458            pygame.display.update()
459
460            # 12 - Slow things down a bit
461            clock.tick(self.fps)
