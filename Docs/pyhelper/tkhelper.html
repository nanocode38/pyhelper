 56def shake_window(
 57    window, dire: Direction = "vertical", jitter_time: Seconds = 2, jitter_count: int = 50, amplitude: px = 1
 58) -> None:
 59    """
 60    Shake the Tkinter window
 61
 62    Args:
 63        window: , needs to have winfo_x(), winfo_y(), update() and geometr() methods
 64        dire: Shaking Direction, can be 'vertical' or 'horizontal'
 65        jitter_time: Total duration of the shaking effect, in Seconds, integer
 66        jitter_count: Number of shakes, up and down count as two, integer
 67        amplitude: Amplitude of the shake, in pixels, integer
 68
 69    Raises:
 70        ValueError: When dire is not 'vertical' or 'horizontal'
 71    """
 72    window.update()
 73    home_position = window.winfo_x(), window.winfo_y()
 74    for i in range(jitter_count):
 75        start_time = time.time()
 76        sign = -1 if i % 2 == 0 else 1
 77        if dire == "horizontal":
 78            window.geometry(f"+{home_position[0]}+{home_position[1] + sign * amplitude}")
 79        elif dire == "vertical":
 80            window.geometry(f"+{home_position[0] + sign * amplitude}+{home_position[1]}")
 81        else:
 82            raise ValueError("Direction must be vertical or horizontal")
 83        while time.time() - start_time < jitter_time / jitter_count:
 84            window.update()
 85    window.geometry(f"+{home_position[0]}+{home_position[1]}")
 86    window.update()

 89def center_window(window, dire: Direction = "all") -> None:
 90    """
 91    Center your Tkinter window
 92
 93
 94    Args:
 95        window: The Tkinter window to be centered
 96        dire: The Direction of centering, can be 'all', 'vertical' or 'horizontal'
 97
 98    Raises:
 99        ValueError: When dire is not any one of 'all', 'vertical' or 'horizontal'
100    """
101    window.update()
102    dire = dire.lower()
103    screen_size = window.winfo_screenwidth(), window.winfo_screenheight()
104    window_size = window.winfo_width(), window.winfo_height()
105    x, y = (screen_size[0] - window_size[0]) // 2, (screen_size[1] - window_size[1]) // 2
106    this_x, this_y = window.winfo_x(), window.winfo_y()
107    if dire == "all":
108        window.geometry(f"+{x}+{y}")
109    elif dire == "vertical":
110        window.geometry(f"+{x}+{this_y}")
111    elif dire == "horizontal":
112        window.geometry(f"+{this_x}+{y}")
113    else:
114        raise ValueError("Direction must be all, vertical or horizontal")
115    window.update()

184def password_window(
185    password: str,
186    title: str = "Password",
187    prompt: str = "Please enter your password",
188    *,
189    mask: str | None = None,
190    error_message: str | None = None,
191    topmost: bool = False,
192    parent: tk.Tk | None = None,
193) -> RetMode:
194    """
195    Create a password input window using Tkinter.
196
197    This function generates a window prompting the user to input a password.
198    It validates the entered password and performs actions based on the correctness of the input.
199
200    Args:
201        password: The correct password string that the user needs to input.
202        title: The title of the password window (default is "Password").
203        prompt: The prompt text displayed in the window (default is "Please input your Password").
204        mask: mask: Must be keyword parameters, Optional character to mask the password input [e.g., '*' or None(for
205        no masking, default)].
206        error_message: Optional custom error message when the password is incorrect.
207                          If not provided, the window will shake to indicate an error.
208        topmost: Boolean indicating whether the window should stay on top of others (default is False).
209        parent: Parent Tkinter window for modal behavior. If None, creates standalone window.
210
211    Returns:
212        Returns RetMode.SUCCESS if the correct password is entered, RetMode.USER_CANCEL if canceled via button,
213        or RetMode.WINDOW_CLOSE if closed via window controls.
214
215    Functionality:
216    - Displays a Tkinter window with a label, an entry field for password input, and two buttons ("OK" and "Cancel").
217    - Validates the entered password against the provided `password`.
218    - If the password is correct, the window closes and returns RetMode.SUCCESS.
219    - If the password is incorrect:
220        - Shakes the window if no custom `error_message` is provided.
221        - Displays an error popup with the custom `error_message` if it is provided.
222    - Centers the window on the screen using the `center_window` function.
223    - Supports optional parameters for customizing the appearance and behavior of the password window.
224    """
225    # Create window hierarchy
226    is_root = parent is None
227    window = tk.Toplevel(parent) if parent else tk.Tk()
228    if is_root:
229        window.title(title)
230
231    # Configure window properties
232    window.wm_attributes("-topmost", int(topmost))
233    window.geometry("300x150")
234    window.resizable(False, False)
235
236    # Status container (using list for mutable closure)
237    result = [RetMode.WINDOW_CLOSE]  # Default: closed without verification
238
239    # Password verification handler
240    def verify_password() -> None:
241        """Validate entered password against reference"""
242        if entry.get() == password:
243            result[0] = RetMode.SUCCESS
244            window.destroy()
245        elif error_message:
246            messagebox.showerror("Verification Error", error_message, parent=window, master=window)
247            entry.select_range(0, tk.END)
248        else:
249            # Calculate shake parameters dynamically
250            shake_window(window, amplitude=min(10, window.winfo_width() // 30))
251            entry.select_range(0, tk.END)
252
253    # Cancel operation handler
254    def cancel_operation() -> None:
255        """Handle explicit cancellation"""
256        result[0] = RetMode.USER_CANCEL
257        window.destroy()
258
259    # Configure UI components
260    ttk.Label(window, text=prompt).pack(pady=(10, 0))
261    entry = ttk.Entry(window, show=mask, width=25)
262    entry.pack(pady=10, padx=20, fill=tk.X)
263    entry.focus_set()
264
265    button_frame = ttk.Frame(window)
266    button_frame.pack(pady=10, fill=tk.X, padx=20)
267
268    ttk.Button(button_frame, text="Cancel", command=cancel_operation).pack(side=tk.LEFT)
269    ttk.Button(button_frame, text="OK", command=verify_password).pack(side=tk.RIGHT)
270
271    # Configure window close protocol
272    window.protocol("WM_DELETE_WINDOW", window.destroy)
273
274    # Bind Enter key to submit
275    window.bind("<Return>", lambda e: verify_password())
276
277    # Position window and start event loop
278    center_window(window)
279    window.transient(parent)
280    window.grab_set()
281    window.wait_window()
282
283    return result[0]

286def setting_password_window(
287    title: str = "Password",
288    prompt: str = "Please input your Password",
289    prompt_again: str = "Please input tour Password again",
290    *,
291    mask: None | str = None,
292    error_message: None | str = None,
293    topmost=False,
294    check_fun: callable = lambda _: True,
295) -> None | str:
296    """
297    Window to set password.
298    Args:
299        title: Main window title
300        prompt: The prompt text for entering password for the first time
301        prompt_again: The prompt text for entering password for the second time
302        mask: Must be keyword parameters, Optional character to mask the password input [e.g., '*' or None(for no masking, default)].
303        error_message: Optional custom error message when the password is incorrect.
304                       If not provided, the window will shake to indicate an error.
305        topmost: Boolean indicating whether the window should stay on top of others (default is False).
306        check_fun: Functions that check whether they comply with password specifications are the default constant true function
307
308    Returns:
309        A string, if the user closes the window (press Cancel or close key, it will return None)
310    """
311    return_val = None
312    widgets_width = []
313
314    window = tk.Tk()
315    window.wm_attributes("-topmost", topmost)
316    window.title(title)
317    tk.Label(window, text=prompt).place(x=10, y=2)
318    entry1 = ttk.Entry(window, show=mask, width=40)
319    entry1.place(x=10, y=30)
320    widgets_width.append(entry1.winfo_reqwidth())
321    tk.Label(window, text=prompt_again).place(x=10, y=60)
322    entry2 = ttk.Entry(window, show=mask, width=40)
323    entry2.place(x=10, y=90)
324    widgets_width.append(entry2.winfo_reqwidth())
325
326    def ok():
327        nonlocal entry1, entry2, return_val, window, error_message
328        if not entry1.get() == entry2.get() or not check_fun(entry1.get()):
329            if not error_message:
330                shake_window(window)
331            else:
332                messagebox.showerror(title=" Password Error", message=error_message, parent=window, master=window)
333            return
334        return_val = entry1.get()
335        window.destroy()
336
337    tmp = ttk.Button(window, text="Cancel", command=lambda: window.destroy())
338    tmp.place(x=105, y=120)
339    widgets_width.append(tmp.winfo_reqwidth())
340    tmp = ttk.Button(window, text="OK", command=ok)
341    tmp.place(x=205, y=120)
342    widgets_width.append(tmp.winfo_reqwidth())
343    window_width = max(widgets_width)
344    window.geometry(f"{window_width + 20}x150")
345    center_window(window)
346    window.wait_window()
347    return return_val

351    """
352    Keep windows updated while the program sleeps
353
354    Args:

357    """

367class Rect:
368    """
369    A Rect class for tkinter
370
371    You can construct a Rect object in the following ways:
372    - `Rect(x, y, width, height)`: Creates a rectangle with its top-left corner at (x, y), a width of `width`, and a height of `height`.
373    - `Rect((x, y), (width, height))`: Creates a rectangle with its top-left corner at (x, y), a width of `width`, and a height of `height`.
374    - `Rect(rect_object)`: Creates a rectangle with the same position and dimensions as the given `rect` object.
375    - `Rect(tk.Widget)`: Generates a Rect object for the specified Widget.
376
377
378    Attributes:
379        x, y(left, top): The x, y position of top and left side
380        w, h(width, height): The width/height of the rect
381        right: The x position of the right side
382        bottom: The y position of the bottom side
383        centerx, centery : The x/y position of the rect center
384        center = (centerx, centery)
385        size = (w, h)
386        topleft = (left, top)
387        topright = (right, top)
388        bottomright = (right, bottom)
389        bottomleft = (left, bottom)
390        midleft = (left, centery)
391        midright = (right, centery)
392        midtop = (centerx, top)
393        midbottom = (centerx, bottom)
394

396        >>> r = Rect(10, 20, 30, 40)

A Rect class for tkinter

You can construct a Rect object in the following ways:

  • Rect(x, y, width, height): Creates a rectangle with its top-left corner at (x, y), a width of width, and a height of height.
  • Rect((x, y), (width, height)): Creates a rectangle with its top-left corner at (x, y), a width of width, and a height of height.
  • Rect(rect_object): Creates a rectangle with the same position and dimensions as the given rect object.
  • Rect(tk.Widget): Generates a Rect object for the specified Widget.
